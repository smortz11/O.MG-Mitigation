1. Follow instructions in the README
2. cd /sys/kernel/config/usb_gadget/<name of the HID>
3. sudo mkdir -p functions/acm.usb0
4. UDC=$(cat UDC)
5. sudo sh -c "echo '' > UDC"
6. sudo ln -s functions/acm.usb0 configs/c.1/
7. sudo sh -c "echo $UDC > UDC"
8. sudo chmod 666 /dev/ttyGS0

ON THE RECEIVING SIDE:
1. sudo usermod -a -G dialout $USER
2. sudo chmod 666 /dev/ttyACM0

NOTE: 
To send from pi to receiver:
receiver: cat /dev/ttyACM0
pi: echo "test" > /dev/ttyGS0

To send from receiver to pi:
pi: cat /dev/ttyGS0
receiver: echo "test" | sudo tee /dev/ttyACM0
